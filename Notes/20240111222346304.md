		? true
		: false;

		if (isSnippetsEnabled) {
			console.log("关闭");
			app.customCss.setCssEnabledStatus(snippetName, false);
			app.customCss.requestLoadSnippets();
		} else {
			console.log("启动");
			app.customCss.setCssEnabledStatus(snippetName, true);
			app.customCss.requestLoadSnippets();
		}
